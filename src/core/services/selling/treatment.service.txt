import { Inject, Injectable } from '@nestjs/common';
import { InjectRepository } from '@nestjs/typeorm';
import { Repository } from 'typeorm';
import { PaginatedService } from '@app/typeorm';
import { REQUEST } from '@nestjs/core';
import { AbstractService } from '../abstract.service';
import { Treatment } from '../../entities/selling/treatment.entity';

@Injectable()
export class TreatmentService extends AbstractService<Treatment> {
  public NOT_FOUND_MESSAGE = `Traitement non trouv√©`;

  constructor(
    @InjectRepository(Treatment)
    private _repository: Repository<Treatment>,
    protected paginatedService: PaginatedService<Treatment>,
    @Inject(REQUEST) protected request: any,
  ) {
    super();
  }

  get repository(): Repository<Treatment> {
    return this._repository;
  }
}
